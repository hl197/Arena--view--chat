# 工具系统参考

## 三层工具体系

```
Tool (抽象基类)
├── 普通 Tool — 单一功能，实现 run() + get_parameters()
├── 可展开 Tool — expandable=True，@tool_action 装饰器自动生成子工具
│   └── AutoGeneratedTool — 从方法签名+docstring 自动提取参数定义
└── 函数 Tool — register_function() 快速注册，自动包装为 ToolResponse
```

## ToolResponse 协议：三种状态

```python
class ToolStatus(Enum):
    SUCCESS = "success"  # 完全按预期执行
    PARTIAL = "partial"  # 结果可用但有折扣（截断、回退）
    ERROR = "error"      # 无有效结果（致命错误）

@dataclass
class ToolResponse:
    status: ToolStatus
    text: str                          # 给 LLM 阅读的格式化文本
    data: Dict[str, Any] = {}          # 结构化数据载荷
    error_info: Optional[Dict] = None  # {"code": "NOT_FOUND", "message": "..."}
    stats: Optional[Dict] = None       # {"time_ms": 50, "api_calls": 1}
    context: Optional[Dict] = None     # {"params_input": {...}, "tool_name": "..."}
```

**便捷构造方法**：
```python
ToolResponse.success(text="完成", data={"result": 42})
ToolResponse.partial(text="部分完成（已截断）", data={"truncated": True})
ToolResponse.error(code="INVALID_PARAM", message="参数不能为空")
```

## Tool 基类设计

```python
class Tool(ABC):
    def __init__(self, name, description, expandable=False):
        self.name = name
        self.description = description
        self.expandable = expandable

    @abstractmethod
    def run(self, parameters: Dict[str, Any]) -> ToolResponse: ...
    @abstractmethod
    def get_parameters(self) -> List[ToolParameter]: ...

    # 自动添加时间统计
    def run_with_timing(self, parameters) -> ToolResponse:
        start = time.time()
        try:
            response = self.run(parameters)
        except Exception as e:
            return ToolResponse.error(code=ToolErrorCode.INTERNAL_ERROR, ...)
        response.stats["time_ms"] = int((time.time() - start) * 1000)
        response.context["params_input"] = parameters
        return response

    # 异步支持
    async def arun(self, parameters) -> ToolResponse: ...      # 在线程池中运行同步
    async def arun_with_timing(self, parameters) -> ToolResponse: ...

    # 转换为 OpenAI function calling schema
    def to_openai_schema(self) -> Dict: ...
```

### 可展开工具（@tool_action 装饰器）

装饰器自动从方法签名和 docstring 提取 name、description、参数类型：

```python
class MemoryTool(Tool):
    def __init__(self):
        super().__init__(name="memory", description="记忆工具集", expandable=True)

    @tool_action("memory_add", "添加新记忆")
    def _add_memory(self, content: str, importance: float = 0.5) -> str:
        """添加记忆
        Args:
            content: 记忆内容
            importance: 重要性分数 (0.0-1.0)
        """
        ...

# AutoGeneratedTool 从方法签名自动提取:
# - 名称: parent.name + method_name (memory_add)
# - 描述: docstring 第一行
# - 参数: 从函数签名提取，required=无默认值
# - 类型映射: str→string, int→integer, float→number, bool→boolean, list→array
```

## ToolRegistry 注册表

```python
class ToolRegistry:
    def __init__(self, circuit_breaker=None):
        self._tools: dict[str, Tool] = {}       # Tool 对象
        self._functions: dict[str, dict] = {}   # 函数工具 {name: {desc, func}}
        self.circuit_breaker = circuit_breaker or CircuitBreaker()

    # 注册 Tool（auto_expand=True 自动展开可展开工具）
    def register_tool(self, tool: Tool, auto_expand=True): ...

    # 注册函数（自动从函数名/docstring 提取名称和描述）
    def register_function(self, func, name=None, description=None): ...

    # 执行工具（含熔断器保护）
    def execute_tool(self, name, input_text) -> ToolResponse:
        if self.circuit_breaker.is_open(name):
            return ToolResponse.error(code="CIRCUIT_OPEN", ...)
        # Tool 对象优先 → 函数工具 → NOT_FOUND
        response = tool.run_with_timing(typed_parameters)
        self.circuit_breaker.record_result(name, response)
        return response

    # 乐观锁元数据缓存
    def cache_read_metadata(self, file_path, metadata): ...
    def get_read_metadata(self, file_path) -> Optional[Dict]: ...
```

## CircuitBreaker 熔断器

**状态机**：`Closed(正常) → [连续失败N次] → Open(熔断) → [等待超时] → Closed(恢复)`

```python
class CircuitBreaker:
    def __init__(self, failure_threshold=3, recovery_timeout=300, enabled=True):
        self.failure_counts: Dict[str, int] = defaultdict(int)
        self.open_timestamps: Dict[str, float] = {}

    def is_open(self, tool_name) -> bool:
        """检查是否熔断；超时自动恢复"""
        if tool_name not in self.open_timestamps: return False
        if time.time() - self.open_timestamps[tool_name] > self.recovery_timeout:
            self.close(tool_name)  # 自动恢复
            return False
        return True

    def record_result(self, tool_name, response):
        if response.status == ToolStatus.ERROR:
            self.failure_counts[tool_name] += 1
            if self.failure_counts[tool_name] >= self.failure_threshold:
                self.open_timestamps[tool_name] = time.time()  # 熔断！
        else:
            self.failure_counts[tool_name] = 0  # 成功则重置

    def get_status(self, tool_name) -> Dict:
        """返回 {state, failure_count, open_since, recover_in_seconds}"""
```

## ToolErrorCode：15 种标准错误码

```python
class ToolErrorCode:
    # 资源相关
    NOT_FOUND = "NOT_FOUND"                # 文件/工具不存在
    ACCESS_DENIED = "ACCESS_DENIED"        # 访问被拒绝
    PERMISSION_DENIED = "PERMISSION_DENIED" # 权限不足
    IS_DIRECTORY = "IS_DIRECTORY"          # 期望文件但得到目录
    BINARY_FILE = "BINARY_FILE"            # 二进制文件无法处理
    # 参数相关
    INVALID_PARAM = "INVALID_PARAM"        # 参数无效/缺失
    INVALID_FORMAT = "INVALID_FORMAT"      # 格式错误
    # 执行相关
    EXECUTION_ERROR = "EXECUTION_ERROR"    # 执行错误
    TIMEOUT = "TIMEOUT"                    # 执行超时
    INTERNAL_ERROR = "INTERNAL_ERROR"      # 内部错误
    # 状态相关
    CONFLICT = "CONFLICT"                  # 乐观锁冲突
    CIRCUIT_OPEN = "CIRCUIT_OPEN"          # 熔断器开启
    # 网络相关
    NETWORK_ERROR = "NETWORK_ERROR"        # 网络失败
    API_ERROR = "API_ERROR"                # API 失败
    RATE_LIMIT = "RATE_LIMIT"              # 速率限制
```

## ToolFilter：子代理工具访问控制

```python
class ToolFilter(ABC):
    @abstractmethod
    def filter(self, all_tools: List[str]) -> List[str]: ...
    @abstractmethod
    def is_allowed(self, tool_name: str) -> bool: ...

class ReadOnlyFilter(ToolFilter):
    """白名单模式：只允许只读工具(Read, LS, Glob, Grep, Skill)"""
    READONLY_TOOLS = {"Read", "ReadTool", "LS", "LSTool", "Glob", "GlobTool",
                       "Grep", "GrepTool", "Skill", "SkillTool"}

class FullAccessFilter(ToolFilter):
    """黑名单模式：排除危险工具(Bash, Terminal, Execute)"""
    DENIED_TOOLS = {"Bash", "BashTool", "Terminal", "TerminalTool",
                     "Execute", "ExecuteTool"}

class CustomFilter(ToolFilter):
    """自定义：白名单模式(只允许listed)或黑名单模式(禁止listed)"""
    def __init__(self, allowed=None, denied=None, mode="whitelist"): ...
```
